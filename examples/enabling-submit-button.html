If you want the submit button to be enabled only after clicking it, use the getSubmitButton() method to check whether or not the submit button is clicked:

{% highlight javascript %} $(document).ready(function() { $('#enableButtonForm') .bootstrapValidator({ ... }) .on('error.field.bv', function(e, data) { if (data.bv.getSubmitButton()) { data.bv.disableSubmitButtons(false); } }) .on('success.field.bv', function(e, data) { if (data.bv.getSubmitButton()) { data.bv.disableSubmitButtons(false); } }); }); {% endhighlight %}
